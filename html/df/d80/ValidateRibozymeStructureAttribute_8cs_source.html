126 public bool matchingAlphaNumerics(string sequenceStructure, string substrateStructure)
127 {
128 string sequenceStructureAlphanumeric = "";

130 Regex r = new Regex(@"[a-z0-9]");
131
132 // Get all alphanumerics in sequence
133 foreach (char c in sequenceStructure)
134 {
135 Match m = r.Match(c.ToString());
136
137 if (m.Success)
138 {
139 sequenceStructureAlphanumeric += c;
140 }
141 }
142
143 // Get all alphanumerics in substrate
144 foreach (char c in substrateStructure)
145 {
146 Match m = r.Match(c.ToString());
147
148 if (m.Success)
149 {
150 substrateStructureAlphanumeric += c;

152 }
153
154 // Sort alphanumeric strings
