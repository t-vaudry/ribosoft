643  private string FormatKey(string unformattedKey)
644  {
645  var result = new StringBuilder();
646  int currentPosition = 0;
647  while (currentPosition + 4 < unformattedKey.Length)
648  {
649  result.Append(unformattedKey.Substring(currentPosition, 4)).Append(" ");
650  currentPosition += 4;
651  }
652  if (currentPosition < unformattedKey.Length)
653  {
654  result.Append(unformattedKey.Substring(currentPosition));
655  }
656 
657  return result.ToString().ToLowerInvariant();
658  }
