491 [HttpPost]

493 public async Task<IActionResult> EnableAuthenticator(EnableAuthenticatorViewModel model)
494 {
495 var user = await _userManager.GetUserAsync(User);
496 if (user == null)
497 {
498 throw new ApplicationException($"Unable to load user with ID '{_userManager.GetUserId(User)}'.");
499 }
500
501 if (!ModelState.IsValid)
502 {
503 await LoadSharedKeyAndQrCodeUriAsync(user, model);
504 return View(model);
505 }
506
507 // Strip spaces and hypens
508 var verificationCode = model.Code.Replace(" ", string.Empty).Replace("-", string.Empty);
509
510 var is2faTokenValid = await _userManager.VerifyTwoFactorTokenAsync(
511 user, _userManager.Options.Tokens.AuthenticatorTokenProvider, verificationCode);
512
513 if (!is2faTokenValid)
514 {
515 ModelState.AddModelError("Code", "Verification code is invalid.");
516 await LoadSharedKeyAndQrCodeUriAsync(user, model);
517 return View(model);
518 }
519
520 await _userManager.SetTwoFactorEnabledAsync(user, true);
521 _logger.LogInformation("User with ID {UserId} has enabled 2FA with an authenticator app.", user.Id);
522 var recoveryCodes = await _userManager.GenerateNewTwoFactorRecoveryCodesAsync(user, 10);
523 TempData[RecoveryCodesKey] = recoveryCodes.ToArray();
524
525 return RedirectToAction(nameof(ShowRecoveryCodes));

643 private string FormatKey(string unformattedKey)
644 {
645 var result = new StringBuilder();
646 int currentPosition = 0;
647 while (currentPosition + 4 < unformattedKey.Length)
648 {
649 result.Append(unformattedKey.Substring(currentPosition, 4)).Append(" ");
650 currentPosition += 4;
651 }
652 if (currentPosition < unformattedKey.Length)
653 {
654 result.Append(unformattedKey.Substring(currentPosition));
655 }
656
657 return result.ToString().ToLowerInvariant();
658 }

666 private string GenerateQrCodeUri(string email, string unformattedKey)
667 {
668 return string.Format(
669 AuthenticatorUriFormat,
670 _urlEncoder.Encode("Ribosoft"),
671 _urlEncoder.Encode(email),
672 unformattedKey);
673 }

680 private async Task LoadSharedKeyAndQrCodeUriAsync(ApplicationUser user, EnableAuthenticatorViewModel model)
681 {
682 var unformattedKey = await _userManager.GetAuthenticatorKeyAsync(user);
683 if (string.IsNullOrEmpty(unformattedKey))
684 {
685 await _userManager.ResetAuthenticatorKeyAsync(user);
686 unformattedKey = await _userManager.GetAuthenticatorKeyAsync(user);
687 }
688
689 model.SharedKey = FormatKey(unformattedKey);
690 model.AuthenticatorUri = GenerateQrCodeUri(user.Email, unformattedKey);
691 }
