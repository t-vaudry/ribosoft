173  private async Task DoStage(Job job, JobState state, Func<Job, bool> acceptFunc, Func<Job, IJobCancellationToken, Task> func, IJobCancellationToken cancellationToken)
174  {
175  if (!acceptFunc(job))
176  {
177  // don't do anything if the stage can't handle this type of job
178  // this also catches errored jobs, etc.
179  return;
180  }
181 
182  cancellationToken.ThrowIfCancellationRequested();
183 
184  // set the job to this stage's state
185  if (job.JobState != state)
186  {
187  job.JobState = state;
188  await _db.SaveChangesAsync();
189  }
190 
191  await func(job, cancellationToken);
192  }

242  {
243  foreach (var ribozymeStructure in job.Ribozyme.RibozymeStructures)
244  {
245  cancellationToken.ThrowIfCancellationRequested();

247  try
248  {
249  // Candidate Generation
250  candidates = candidateGenerator.GenerateCandidates(
251  ribozymeStructure.Sequence,

254  ribozymeStructure.SubstrateStructure,

256  }

260  job.StatusMessage = e.Message;

264  }

266  // Algorithms
267  try
268  {
269  uint batchCount = 0;
270  _db.ChangeTracker.AutoDetectChangesEnabled = false;
271 
272  foreach (var candidate in candidates)
273  {
274  cancellationToken.ThrowIfCancellationRequested();
275  RunScoreAlgorithms(candidate, job, ribozymeStructure);
276 
277  if (++batchCount % 100 == 0)
278  {
279  await _db.SaveChangesAsync();
280  await RecreateDbContext();
281  _db.ChangeTracker.AutoDetectChangesEnabled = false;
282  batchCount = 0;
283  }
284  }
285 
286  await RecreateDbContext();
287  }

293  return;
294  }

298  _db.Jobs.Attach(job);
299  await _db.SaveChangesAsync();

303  }
304  }

433  private async Task RunBlast(Job job, IJobCancellationToken cancellationToken)
434  {
435  // check if blastn is available; if it isn't, ignore specificity
436  if (!_blaster.IsAvailable())
437  {
438  _logger.LogWarning("RibosoftWarning | BLAST Service is not available!!");
439  return;
440  }
441 
442  var designs = _db.Designs
443  .Where(d => d.JobId == job.Id)
444  .AsEnumerable()
445  .GroupBy(d => new { d.CutsiteIndex, d.SubstrateSequenceLength });
446 
447  foreach (var designGroup in designs)
448  {
449  cancellationToken.ThrowIfCancellationRequested();
450 
451  var design = designGroup.First();
452  var substrateSequence = job.RNAInput.Substring(design.CutsiteIndex, design.SubstrateSequenceLength);
453 
454  if (string.IsNullOrEmpty(substrateSequence))
455  {
456  continue;
457  }
458 
459  // calculate the substrate specificity score, which is common to all designs in this group
460  var substrateSpecificityScore = CalculateSpecificity(substrateSequence, job.Assembly.Path);
461 
462  if (job.SpecificityMethod == SpecificityMethod.CleavageAndHybridization)
463  {
464  // if we're doing hybridization specificity, also compute the score for the design sequence complement
465  foreach (var d in designGroup)
466  {
467  d.SpecificityScore = substrateSpecificityScore + CalculateSpecificity(new Sequence(d.Sequence).GetComplement(), job.Assembly.Path);
468  }
469  }
470  else
471  {
472  // for cleavage-only specificity, only use the substrate specificity score
473  foreach (var d in designGroup)
474  {
475  d.SpecificityScore = substrateSpecificityScore;
476  }
477  }
478  }
479 
480  // Specificity is minimized to 1
481  // Anything below 1 means there is absolutely no matching in the organism and will not bond
482  // Therefore, remove the design
483  _db.Designs.RemoveRange(_db.Designs.Where(d => d.SpecificityScore < 1.0f));
484 
485  var completedDesigns = _db.Designs.Where(d => d.JobId == job.Id);
486 
487  // Check that there are designs left
488  if (!completedDesigns.Any())
489  {
490  job.JobState = JobState.Warning;
491  job.StatusMessage = "No designs returned from Candidate Generation!";
492  _logger.LogError("No designs returned from Candidate Generation!");
493  _db.Jobs.Attach(job);
494  await _db.SaveChangesAsync();
495  return;
496  }
497 
498  float deltaSpecificity = completedDesigns.Max(d => d.SpecificityScore.GetValueOrDefault()) - completedDesigns.Min(d => d.SpecificityScore.GetValueOrDefault());
499  job.SpecificityTolerance *= deltaSpecificity;
500  _db.Jobs.Attach(job);
501 
502  await _db.SaveChangesAsync();
503  }

511  private float CalculateSpecificity(string sequence, string database)
512  {
513  var specificityScore = 0.0f;
514  var blastParameters = BlastParametersForQuery(database, sequence);
515  blastParameters.Query = sequence;
516  var output = _blaster.Run(blastParameters);
517 
518  foreach (var line in output.Split(new[] {Environment.NewLine}, StringSplitOptions.RemoveEmptyEntries))
519  {
520  var fields = line.Split('\t');
521 
522  // filter out X_ predictions
523  //if (fields[1].StartsWith('X'))
524  //{
525  // continue;
526  //}
527 
528  specificityScore += float.Parse(fields[2]) * float.Parse(fields[3]) / 10000;
529  }
530 
531  return specificityScore;
532  }
