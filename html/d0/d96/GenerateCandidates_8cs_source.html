96 [AutomaticRetry(Attempts = 0)]
97 public async Task Phase1(int jobId, IJobCancellationToken cancellationToken)
98 {
99 var job = GetJob(jobId);
100
101 // TODO - temporarily catch retried jobs
102 await DoStage(job, JobState.Errored, j => j.JobState != JobState.New, async (j, c) => { await Task.CompletedTask; }, cancellationToken);
103
104 // run candidate generator
105 await DoStage(job, JobState.CandidateGenerator, j => j.JobState == JobState.New, RunCandidateGenerator, cancellationToken);
106
107 // calculate structure score
108 await DoStage(job, JobState.Structure, j => j.JobState == JobState.CandidateGenerator, CalculateStructure, cancellationToken);
109
110 // queue phase 2 job for in-vivo runs (blast)
111 await DoStage(job, JobState.QueuedPhase2, j => j.JobState == JobState.Structure && j.TargetEnvironment == TargetEnvironment.InVivo, async (j, c) =>
112 {
113 BackgroundJob.Enqueue<GenerateCandidates>(x => x.Phase2(j.Id, c));
114 await Task.CompletedTask;
115 }, cancellationToken);
116
117 // queue phase 3 job for in-vitro runs, skipping phase 2 (MOO)
118 await DoStage(job, JobState.QueuedPhase3, j => j.JobState == JobState.Structure && j.TargetEnvironment == TargetEnvironment.InVitro, async (j, c) =>
119 {
120 BackgroundJob.Enqueue<GenerateCandidates>(x => x.Phase3(j.Id, c));
121 await Task.CompletedTask;
122 }, cancellationToken);
123 }

181 private async Task DoStage(Job job, JobState state, Func<Job, bool> acceptFunc, Func<Job, IJobCancellationToken, Task> func, IJobCancellationToken cancellationToken)
182 {
183 if (!acceptFunc(job))
184 {
185 // don't do anything if the stage can't handle this type of job
186 // this also catches errored jobs, etc.
187 return;
188 }
189
190 cancellationToken.ThrowIfCancellationRequested();
191
192 // set the job to this stage's state
193 if (job.JobState != state)
194 {
195 job.JobState = state;
196 await _db.SaveChangesAsync();
197 }
198
199 await func(job, cancellationToken);
200 }

233 private async Task RunCandidateGenerator(Job job, IJobCancellationToken cancellationToken)
234 {
235 List<string> rnaInputs = new List<string>();
236 if (job.FivePrime || job.OpenReadingFrame || job.ThreePrime)
237 {
238 SetTargetRegions(job, ref rnaInputs);
239 }
240 else
241 {
242 job.JobState = JobState.Warning;
243 job.StatusMessage = "No Target Region Selected!";
244 await _db.SaveChangesAsync();
245 return;
246 }
247
248 CandidateGeneration.CandidateGenerator candidateGenerator = new CandidateGeneration.CandidateGenerator();
249 foreach (var rnaInput in rnaInputs)
250 {
251 RNAStructure = _ribosoftAlgo.MFEFold(rnaInput);
252
253 foreach (var ribozymeStructure in job.Ribozyme.RibozymeStructures)
254 {
255 cancellationToken.ThrowIfCancellationRequested();
256 IEnumerable<Candidate> candidates;
257 try
258 {
259 // Candidate Generation
260 candidates = candidateGenerator.GenerateCandidates(
261 ribozymeStructure.Sequence,
262 ribozymeStructure.Structure,
263 ribozymeStructure.SubstrateTemplate,
264 ribozymeStructure.SubstrateStructure,
265 rnaInput);
266 }
267 catch (CandidateGeneration.CandidateGenerationException e)
268 {
269 job.JobState = JobState.Errored;
270 job.StatusMessage = e.Message;
271 _logger.LogError(e, "Exception occurred during Candidate Generation.");
272 await _db.SaveChangesAsync();
273 return;
274 }
275
276 // Algorithms
277 try
278 {
279 uint batchCount = 0;
280 _db.ChangeTracker.AutoDetectChangesEnabled = false;
281
282 foreach (var candidate in candidates)
283 {
284 cancellationToken.ThrowIfCancellationRequested();
285 RunScoreAlgorithms(candidate, job, ribozymeStructure, RNAStructure);
286
287 if (++batchCount % 100 == 0)
288 {
289 await _db.SaveChangesAsync();
290 await RecreateDbContext();
291 _db.ChangeTracker.AutoDetectChangesEnabled = false;
292 batchCount = 0;
293 }
294 }
295
296 await RecreateDbContext();
297 }
298 catch (RibosoftAlgoException e)
299 {
300 job.JobState = JobState.Errored;
301 job.StatusMessage = e.Code.ToString();
302 _logger.LogError(e, "Exception occurred during Ribosoft Algorithms.");
303 return;
304 }
305 finally
306 {
307 _db.ChangeTracker.AutoDetectChangesEnabled = true;
308 _db.Jobs.Attach(job);
309 await _db.SaveChangesAsync();
310 }
311
312 candidateGenerator.Clear();
313 }
314 }
315
316 _db.ChangeTracker.AutoDetectChangesEnabled = false;
317 var designs = _db.Designs.Where(j => j.JobId == job.Id).ToList();
318
319 // Check that there are designs left
320 if (!designs.Any())
321 {
322 job.JobState = JobState.Warning;
323 job.StatusMessage = "No designs returned from Candidate Generation!";
324 _logger.LogError("No designs returned from Candidate Generation!");
325 _db.ChangeTracker.AutoDetectChangesEnabled = true;
326 _db.Jobs.Attach(job);
327 await _db.SaveChangesAsync();
328 return;
329 }
330
331 job.DesiredTempTolerance *= designs.Max(d => d.DesiredTemperatureScore.GetValueOrDefault()) - designs.Min(d => d.DesiredTemperatureScore.GetValueOrDefault());
332 job.AccessibilityTolerance *= designs.Max(d => d.AccessibilityScore.GetValueOrDefault()) - designs.Min(d => d.AccessibilityScore.GetValueOrDefault());
333
334 _db.ChangeTracker.AutoDetectChangesEnabled = true;
335 _db.Jobs.Attach(job);
336 await _db.SaveChangesAsync();
337 }

344 private void SetTargetRegions(Job job, ref List<string> rnaInputs)
345 {
346 if (job.FivePrime && job.OpenReadingFrame && job.ThreePrime)
347 {
348 rnaInputs.Add(job.RNAInput);
349 }
350 else if (job.FivePrime && job.OpenReadingFrame)
351 {
352 rnaInputs.Add(job.RNAInput.Substring(0, job.OpenReadingFrameEnd));
353 }
354 else if (job.OpenReadingFrame && job.ThreePrime)
355 {
356 rnaInputs.Add(job.RNAInput.Substring(job.OpenReadingFrameStart, job.RNAInput.Length - job.OpenReadingFrameStart - 1));
357 }
358 else if (job.FivePrime && job.ThreePrime)
359 {
360 rnaInputs.Add(job.RNAInput.Substring(0, job.OpenReadingFrameStart));
361 rnaInputs.Add(job.RNAInput.Substring(job.OpenReadingFrameEnd, job.RNAInput.Length - job.OpenReadingFrameEnd - 1));
362 }
363 else if (job.FivePrime)
364 {
365 rnaInputs.Add(job.RNAInput.Substring(0, job.OpenReadingFrameStart));
366 }
367 else if (job.OpenReadingFrame)
368 {
369 rnaInputs.Add(job.RNAInput.Substring(job.OpenReadingFrameStart, job.OpenReadingFrameEnd - job.OpenReadingFrameStart - 1));
370 }
371 else if (job.ThreePrime)
372 {
373 rnaInputs.Add(job.RNAInput.Substring(job.OpenReadingFrameEnd, job.RNAInput.Length - job.OpenReadingFrameEnd - 1));
374 }

383 private void RunScoreAlgorithms(Candidate candidate, Job job, RibozymeStructure ribozymeStructure, string RNAStructure)
384 {
385 var idealStructurePattern = new Regex(@"[^.^(^)]");
386 string ideal = idealStructurePattern.Replace(candidate.Structure ?? string.Empty, ".");
387
388 float naConcentration = job.Na.GetValueOrDefault();
389 float probeConcentration = job.Probe.GetValueOrDefault();
390 float targetTemperature = job.TargetTemperature.GetValueOrDefault();
391
392 var temperatureScore = _ribosoftAlgo.Anneal(candidate, candidate.SubstrateSequence ?? string.Empty,
393 candidate.SubstrateStructure ?? string.Empty, naConcentration, probeConcentration, targetTemperature);
394
395 foreach (var cutsiteIndex in candidate.CutsiteIndices ?? new List<int>())
396 {
397 var accessibilityScore = _ribosoftAlgo.Accessibility(candidate, RNAStructure,
398 cutsiteIndex, naConcentration, probeConcentration, targetTemperature);
399
400 _db.Designs.Add(new Design
401 {
402 JobId = job.Id,
403
404 Sequence = candidate.Sequence?.GetString() ?? string.Empty,
405 IdealStructure = ideal,
406 SubstrateSequence = candidate.SubstrateSequence ?? "",
407
408 // TODO: save actual cutsite (cutsiteIndex + ribozymeStructure.Cutsite + candidate.CutsiteNumberOffset)
409 CutsiteIndex = cutsiteIndex,
410
411 SubstrateSequenceLength = candidate.SubstrateSequence?.Length ?? 0,
412 AccessibilityScore = accessibilityScore,
413 DesiredTemperatureScore = temperatureScore
414 });
415 }
416 }

470 private async Task RunBlast(Job job, IJobCancellationToken cancellationToken)
471 {
472 // check if blastn is available; if it isn't, ignore specificity
473 if (!_blaster.IsAvailable())
474 {
475 _logger.LogWarning("RibosoftWarning | BLAST Service is not available!!");
476 return;
477 }
478
479 var designs = _db.Designs
480 .Where(d => d.JobId == job.Id)
481 .AsEnumerable()
482 .GroupBy(d => new { d.CutsiteIndex, d.SubstrateSequence });
483
484 foreach (var designGroup in designs)
485 {
486 cancellationToken.ThrowIfCancellationRequested();
487
488 var design = designGroup.First();
489 var substrateSequence = design.SubstrateSequence;
490
491 if (string.IsNullOrEmpty(substrateSequence))
492 {
493 continue;
494 }
495
496 // calculate the substrate specificity score, which is common to all designs in this group
497 var substrateSpecificityScore = CalculateSpecificity(substrateSequence, job.Assembly.Path);
498
499 foreach (var d in designGroup)
500 {
501 d.SpecificityScore = substrateSpecificityScore;
502 }
503 }
504
505 // Specificity is minimized to 1 in the case of a wildtype gene (a score of 0 is ideal for synthetic genes)
506 // Anything below 1 means there is absolutely no matching in the organism and will not bond
507 // Therefore, remove the design
508 if (job.SpecificityMethod == SpecificityMethod.Wildtype)
509 {
510 _db.Designs.RemoveRange(_db.Designs.Where(d => d.SpecificityScore < 1.0f));
511 }
512
513 var completedDesigns = _db.Designs.Where(d => d.JobId == job.Id);
514
515 // Check that there are designs left
516 if (!completedDesigns.Any())
517 {
518 job.JobState = JobState.Warning;
519 job.StatusMessage = "No designs returned from Candidate Generation!";
520 _logger.LogError("No designs returned from Candidate Generation!");
521 _db.Jobs.Attach(job);
522 await _db.SaveChangesAsync();
523 return;
524 }
525
526 float deltaSpecificity = completedDesigns.Max(d => d.SpecificityScore.GetValueOrDefault()) - completedDesigns.Min(d => d.SpecificityScore.GetValueOrDefault());
527 job.SpecificityTolerance *= deltaSpecificity;
528 _db.Jobs.Attach(job);
529
530 await _db.SaveChangesAsync();
531 }

539 private float CalculateSpecificity(string sequence, string database)
540 {
541 var specificityScore = 0.0f;
542 var blastParameters = BlastParametersForQuery(database, sequence);
543 blastParameters.Query = sequence;
544 var output = _blaster.Run(blastParameters);
545
546 foreach (var line in output.Split(new[] {Environment.NewLine}, StringSplitOptions.RemoveEmptyEntries))
547 {
548 var fields = line.Split('\t');
549
550 // filter out X_ predictions
551 //if (fields[1].StartsWith('X'))
552 //{
553 // continue;
554 //}
555
556 specificityScore += float.Parse(fields[2]) * float.Parse(fields[3]) / 10000;
557 }
558
559 return specificityScore;
560 }

568 private BlastParameters BlastParametersForQuery(string database, string query)
569 {
570 var blastParameters = new BlastParameters
571 {
572 BlastDbPath = _configuration.GetValue("Blast:BLASTDB", string.Empty) ?? string.Empty,
573 Database = database,
574 UseIndex = true,
575 LowercaseMasking = true,
576 OutputFormat = "6 qseqid saccver pident qcovs",
577 NumThreads = _configuration.GetValue("Blast:NumThreads", 4),
578 MaxTargetSequences = 200
579 };
580
581 if (query.Length <= 30)
582 {
583 // adjust parameters for a short query (as NCBI does it)
584 blastParameters.Task = BlastParameters.BlastTask.blastn_short;
585 blastParameters.ExpectValue = 1000.0f;
586 }
587
588 return blastParameters;
