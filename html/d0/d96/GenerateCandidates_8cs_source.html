181  private async Task DoStage(Job job, JobState state, Func<Job, bool> acceptFunc, Func<Job, IJobCancellationToken, Task> func, IJobCancellationToken cancellationToken)
182  {
183  if (!acceptFunc(job))
184  {
185  // don't do anything if the stage can't handle this type of job
186  // this also catches errored jobs, etc.
187  return;
188  }
189 
190  cancellationToken.ThrowIfCancellationRequested();
191 
192  // set the job to this stage's state
193  if (job.JobState != state)
194  {
195  job.JobState = state;
196  await _db.SaveChangesAsync();
197  }
198 
199  await func(job, cancellationToken);
200  }

286

470  private async Task RunBlast(Job job, IJobCancellationToken cancellationToken)
471  {
472  // check if blastn is available; if it isn't, ignore specificity
473  if (!_blaster.IsAvailable())
474  {
475  _logger.LogWarning("RibosoftWarning | BLAST Service is not available!!");
476  return;
477  }
478 
479  var designs = _db.Designs
480  .Where(d => d.JobId == job.Id)
481  .AsEnumerable()
482  .GroupBy(d => new { d.CutsiteIndex, d.SubstrateSequence });
483 
484  foreach (var designGroup in designs)
485  {
486  cancellationToken.ThrowIfCancellationRequested();
487 
488  var design = designGroup.First();
489  var substrateSequence = design.SubstrateSequence;
490 
491  if (string.IsNullOrEmpty(substrateSequence))
492  {
493  continue;
494  }
495 
496  // calculate the substrate specificity score, which is common to all designs in this group
497  var substrateSpecificityScore = CalculateSpecificity(substrateSequence, job.Assembly.Path);
498 
499  foreach (var d in designGroup)
500  {
501  d.SpecificityScore = substrateSpecificityScore;
502  }
503  }
504 
505  // Specificity is minimized to 1 in the case of a wildtype gene (a score of 0 is ideal for synthetic genes)
506  // Anything below 1 means there is absolutely no matching in the organism and will not bond
507  // Therefore, remove the design
508  if (job.SpecificityMethod == SpecificityMethod.Wildtype)
509  {
510  _db.Designs.RemoveRange(_db.Designs.Where(d => d.SpecificityScore < 1.0f));
511  }
512 
513  var completedDesigns = _db.Designs.Where(d => d.JobId == job.Id);
514 
515  // Check that there are designs left
516  if (!completedDesigns.Any())
517  {
518  job.JobState = JobState.Warning;
519  job.StatusMessage = "No designs returned from Candidate Generation!";
520  _logger.LogError("No designs returned from Candidate Generation!");
521  _db.Jobs.Attach(job);
522  await _db.SaveChangesAsync();
523  return;
524  }
525 
526  float deltaSpecificity = completedDesigns.Max(d => d.SpecificityScore.GetValueOrDefault()) - completedDesigns.Min(d => d.SpecificityScore.GetValueOrDefault());
527  job.SpecificityTolerance *= deltaSpecificity;
528  _db.Jobs.Attach(job);
529 
530  await _db.SaveChangesAsync();
531  }

539  private float CalculateSpecificity(string sequence, string database)
540  {
541  var specificityScore = 0.0f;
542  var blastParameters = BlastParametersForQuery(database, sequence);
543  blastParameters.Query = sequence;
544  var output = _blaster.Run(blastParameters);
545 
546  foreach (var line in output.Split(new[] {Environment.NewLine}, StringSplitOptions.RemoveEmptyEntries))
547  {
548  var fields = line.Split('\t');
549 
550  // filter out X_ predictions
551  //if (fields[1].StartsWith('X'))
552  //{
553  // continue;
554  //}
555 
556  specificityScore += float.Parse(fields[2]) * float.Parse(fields[3]) / 10000;
557  }
558 
559  return specificityScore;
560  }
