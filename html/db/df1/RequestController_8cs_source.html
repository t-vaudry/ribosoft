190 private async Task<bool> ExceededMaxRequests(ApplicationUser user)
191 {
192 if ((await _userManager.GetRolesAsync(user)).ToList().Contains("Administrator"))
193 {
194 return await _context.Jobs.CountAsync(j => j.OwnerId == user.Id) >= 100;
195 }
196 else
197 {
198 return await _context.Jobs.CountAsync(j => j.OwnerId == user.Id) >= 20;
