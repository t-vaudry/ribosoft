26 public IList<T> Optimize<T>(IList<T> candidates, int rank) where T : class, IRankable<OptimizeItem<float>>
27 {
28 // If candidates are empty, return
29 if (candidates.Count == 0) {
30 throw new MultiObjectiveOptimizationException(R_STATUS.R_EMPTY_CANDIDATE_LIST, "List of Candidates is empty!");
31 }
32
33 List<T> rankedCandidates = new List<T>();
34
35 // List for the current rank
36 IList<T> frontCandidates = new List<T>();
37
38 foreach (var victim in candidates) {
39 var dominated = candidates.Where(candidate => candidate != victim).Any(dominator => ParetoDominate(victim, dominator));
40
41 // If not dominated by any other candidate, add to Pareto Front
42 if (!dominated) {
43 frontCandidates.Add(victim);
44 }
45 }
46
47 // Rank non-dominated candidates, and remove from list
48 // If all candidates are dominated, rank them all equally
49 // They will be reranked after using partial dominance
50 if (frontCandidates.Any()) {
51 foreach (var rankedCandidate in frontCandidates) {
52 rankedCandidate.Rank = rank;
53 rankedCandidates.Add(rankedCandidate);
54 candidates.Remove(rankedCandidate);
55 }
56 } else {
57 foreach (var candidate in candidates) {
58 candidate.Rank = rank;
59 rankedCandidates.Add(candidate);
60 candidates.Remove(candidate);
61 }
62 }
63
64 // Try to rerank equally ranked candidates. If a candidate beats another candidate in more fitness categories, it will be ranked higher
65 // If candidate A beats candidate B in more categories, candidate B beats candidate C and candidate C beats candidate A, they will be ranked equally
66 rankedCandidates = UpdateRank(rankedCandidates, ref rank);
67
68 // Recursively call function to continue ranking
69 if (candidates.Any()) {
70 rankedCandidates.AddRange(Optimize(candidates, rank));
71 }
72
73 return rankedCandidates;

85 private bool ParetoDominate(IRankable<OptimizeItem<float>> victim, IRankable<OptimizeItem<float>> dominator)
86 {
87 bool dominated = true;
88 bool strictlyDominated = false;
89
90 using (var dominatorEnumerator = dominator.Comparables.GetEnumerator())
91 {
92 foreach (var v in victim.Comparables)
93 {
94 // If two candidates have different number of fitness values, something went wrong
95 if (!dominatorEnumerator.MoveNext())
96 {
97 throw new MultiObjectiveOptimizationException(R_STATUS.R_FITNESS_VALUE_LENGTHS_DIFFER, "Candidates have different number of fitness values!");
98 }
99
100 var d = dominatorEnumerator.Current;
101
102 if (((d.Type == OptimizeType.MIN) && (d.Value <= v.Value)) || (((d.Type == OptimizeType.MAX)) && (d.Value >= v.Value)))
103 {
104 if (Math.Abs(d.Value - v.Value) > v.Tolerance)
105 {
106 strictlyDominated = true;
107 }
108 }
109 else
110 {
111 dominated = false;
112 break;
113 }
114 }
115 }
116
117 return dominated && strictlyDominated;
118 }

127 public List<T> UpdateRank<T>(List<T> candidates, ref int rank) where T : class, IRankable<OptimizeItem<float>>
128 {
129 List<T> rankedCandidates = new List<T>();
130
131 // List for the current rank
132 List<T> frontCandidates = new List<T>();
133
134 foreach (var victim in candidates)
135 {
136 var partiallyDominant = candidates.Where(candidate => candidate != victim).Any(dominator => PartialDominate(victim, dominator));
137
138 // If not dominated by any other candidate, add to Partial Dominance Front
139 if (partiallyDominant)
140 {
141 frontCandidates.Add(victim);
142 }
143 }
144
145 // Rank non-dominated candidates, and remove from list
146 // If all candidates are dominated, rank them all equally
147 if (frontCandidates.Any())
148 {
149 // If the number of initial candidates and the number of front candidates are the same, we have a circular "rank" (A beats B beats C beats A)
150 // Assign them the same rank
151 if (candidates.Count() == frontCandidates.Count())
152 {
153 foreach (var rankedCandidate in frontCandidates)
154 {
155 rankedCandidate.Rank = rank;
156 rankedCandidates.Add(rankedCandidate);
157 }
158 rank++;
159 }
160 else
161 {
162 foreach (var rankedCandidate in frontCandidates)
163 {
164 candidates.Remove(rankedCandidate);
165 }
166
167 if (frontCandidates.Count() == 1)
168 {
169 // If only one candidate is in the front, skip the function call and simply assign it the current rank.
170 frontCandidates[0].Rank = rank;
171 rankedCandidates.Add(frontCandidates[0]);
172 rank++;
173 }
174 else
175 {
176 // Recursively call function to continue ranking
177 rankedCandidates.AddRange(UpdateRank(frontCandidates, ref rank));
178 }
179
180 if (candidates.Any())
181 {
182 // If only one candidate is left, skip the function call and simply assign it the current rank.
183 if (candidates.Count() == 1)
184 {
185 candidates[0].Rank = rank;
186 rankedCandidates.Add(candidates[0]);
187 rank++;
188 }
189 else
190 {
191 // Recursively call function to continue ranking
192 rankedCandidates.AddRange(UpdateRank(candidates, ref rank));
193 }
194 }
195 }
196 }
197 else
198 {
199 foreach (var candidate in candidates)
200 {
201 candidate.Rank = rank;
202 rankedCandidates.Add(candidate);
203 }
204 rank++;
205 }
206

218 private bool PartialDominate(IRankable<OptimizeItem<float>> victim, IRankable<OptimizeItem<float>> dominator)
219 {
220 int victimScore = 0;
221 int dominatorScore = 0;
222
223 using (var dominatorEnumerator = dominator.Comparables.GetEnumerator())
224 {
225 foreach (var v in victim.Comparables)
226 {
227 // If two candidates have different number of fitness values, something went wrong
228 if (!dominatorEnumerator.MoveNext())
229 {
230 throw new MultiObjectiveOptimizationException(R_STATUS.R_FITNESS_VALUE_LENGTHS_DIFFER, "Candidates have different number of fitness values!");
231 }
232
233 var d = dominatorEnumerator.Current;
234
235 if (d.Value == v.Value)
236 continue;
237
238 if (((d.Type == OptimizeType.MIN) && (d.Value < v.Value)) || (((d.Type == OptimizeType.MAX)) && (d.Value > v.Value)))
239 {
240 dominatorScore++;
241 }
242 else
243 {
244 victimScore++;
245 }
246 }
247 }
248
249 return (victimScore > dominatorScore);
250 }
