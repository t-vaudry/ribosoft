115 string rettypeFastaString = "rettype=fasta";
116 string retmodeString = "retmode=text";
117 var fastaRequestURL = string.Concat(genbankHostString, "?", idString, "&", databaseString, "&", reportString, "&", fmtmaskString, "&",
118 maxdownloadString, "&", rettypeFastaString, "&", retmodeString);
119
120 var fastaResult = await client.GetAsync(fastaRequestURL);
121 fastaResult.EnsureSuccessStatusCode();
122 var fastaResponseString = await fastaResult.Content.ReadAsStringAsync();
123
124 var sequenceIndex = fastaResponseString.IndexOf("\n") + 1;
125 if (sequenceIndex == -1)
126 {
127 throw new GenbankRequestsException("Genbank response was empty");
128 }
129
130 var sequenceString = fastaResponseString.Substring(sequenceIndex);
131 sequenceString = sequenceString.Replace("\n", "");
132 sequenceString = sequenceString.Replace("T", "U");
133
134 var match = Regex.Match(sequenceString, @"[^ACGU]");
135 if (match.Success)
136 {
137 throw new GenbankRequestsException(String.Format("A non-base character {0} was found in the sequence retrieved.", match.Value));
138 }
139
140 return sequenceString;
