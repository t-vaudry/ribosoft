92 public async Task<IActionResult> Login(LoginViewModel model, string? returnUrl = null)
93 {
94 ViewData["ReturnUrl"] = returnUrl;
95 if (ModelState.IsValid)
96 {
97 // This doesn't count login failures towards account lockout
98 // To enable password failures to trigger account lockout, set lockoutOnFailure: true
99 var result = await _signInManager.PasswordSignInAsync(model.Email, model.Password, model.RememberMe, lockoutOnFailure: false);
100 if (result.Succeeded)
101 {
102 _logger.LogInformation("User logged in.");
103 return RedirectToLocal(returnUrl);
104 }
105 if (result.RequiresTwoFactor)
106 {
107 return RedirectToAction(nameof(LoginWith2fa), new { returnUrl, model.RememberMe });
108 }
109 if (result.IsLockedOut)
110 {
111 _logger.LogWarning("User account locked out.");
112 return RedirectToAction(nameof(Lockout));
113 }
114 else
115 {
116 ModelState.AddModelError(string.Empty, "Invalid login attempt.");
117 return View(model);
118 }
119 }
120
121 // If we got this far, something failed, redisplay form
122 return View(model);

182 [HttpPost]
183 [AllowAnonymous]
184 [ValidateAntiForgeryToken]
185 public async Task<IActionResult> LoginWith2fa(LoginWith2faViewModel model, bool rememberMe, string? returnUrl = null)
186 {
187 if (!ModelState.IsValid)
188 {
189 return View(model);
190 }
191
192 var user = await _signInManager.GetTwoFactorAuthenticationUserAsync();
193 if (user == null)
194 {
195 throw new ApplicationException($"Unable to load user with ID '{_userManager.GetUserId(User)}'.");
196 }
197
198 var authenticatorCode = model.TwoFactorCode.Replace(" ", string.Empty).Replace("-", string.Empty);
199
200 var result = await _signInManager.TwoFactorAuthenticatorSignInAsync(authenticatorCode, rememberMe, model.RememberMachine);
201
202 if (result.Succeeded)
203 {
204 _logger.LogInformation("User with ID {UserId} logged in with 2fa.", user.Id);
205 return RedirectToLocal(returnUrl);
206 }
207 else if (result.IsLockedOut)
208 {
209 _logger.LogWarning("User with ID {UserId} account locked out.", user.Id);
210 return RedirectToAction(nameof(Lockout));
211 }
212 else
213 {
214 _logger.LogWarning("Invalid authenticator code entered for user with ID {UserId}.", user.Id);
215 ModelState.AddModelError(string.Empty, "Invalid authenticator code.");
216 return View();
217 }

384 [HttpGet]
385 [AllowAnonymous]
386 public async Task<IActionResult> ExternalLoginCallback(string? returnUrl = null, string? remoteError = null)
387 {
388 if (remoteError != null)
389 {
390 ErrorMessage = $"Error from external provider: {remoteError}";
391 return RedirectToAction(nameof(Login));
392 }
393 var info = await _signInManager.GetExternalLoginInfoAsync();
394 if (info == null)
395 {
396 return RedirectToAction(nameof(Login));
397 }
398
399 // Sign in the user with this external login provider if the user already has a login.
400 var result = await _signInManager.ExternalLoginSignInAsync(info.LoginProvider, info.ProviderKey, isPersistent: false, bypassTwoFactor: true);
401 if (result.Succeeded)
402 {
403 _logger.LogInformation("User logged in with {Name} provider.", info.LoginProvider);
404 return RedirectToLocal(returnUrl);
405 }
406 if (result.IsLockedOut)
407 {
408 return RedirectToAction(nameof(Lockout));
409 }
410 else
411 {
412 // If the user does not have an account, then ask the user to create an account.
413 ViewData["ReturnUrl"] = returnUrl;
414 ViewData["LoginProvider"] = info.LoginProvider;
415 var email = info.Principal.FindFirstValue(ClaimTypes.Email);
416 return View("ExternalLogin", new ExternalLoginViewModel { Email = email });
417 }
418 }

429 public async Task<IActionResult> ExternalLoginConfirmation(ExternalLoginViewModel model, string? returnUrl = null)
430 {
431 if (ModelState.IsValid)
432 {
433 // Get the information about the user from the external login provider
434 var info = await _signInManager.GetExternalLoginInfoAsync();
435 if (info == null)
436 {
437 throw new ApplicationException("Error loading external login information during confirmation.");
438 }
439 var user = new ApplicationUser { UserName = model.Email, Email = model.Email };
440 var result = await _userManager.CreateAsync(user);
441 if (result.Succeeded)
442 {
443 result = await _userManager.AddLoginAsync(user, info);
444 if (result.Succeeded)
445 {
446 await _signInManager.SignInAsync(user, isPersistent: false);
447 _logger.LogInformation("User created an account using {Name} provider.", info.LoginProvider);
448 return RedirectToLocal(returnUrl);
449 }
450 }
451 AddErrors(result);
452 }

500 public async Task<IActionResult> ForgotPassword(ForgotPasswordViewModel model)
501 {
502 if (ModelState.IsValid)
503 {
504 var user = await _userManager.FindByEmailAsync(model.Email);
505 if (user == null || !(await _userManager.IsEmailConfirmedAsync(user)))
506 {
507 // Don't reveal that the user does not exist or is not confirmed
508 return RedirectToAction(nameof(ForgotPasswordConfirmation));
509 }
510
511 // For more information on how to enable account confirmation and password reset please
512 // visit https://go.microsoft.com/fwlink/?LinkID=532713
513 var code = await _userManager.GeneratePasswordResetTokenAsync(user);
514 var callbackUrl = Url.ResetPasswordCallbackLink(user.Id, code, Request.Scheme);
515 await _emailSender.SendEmailAsync(model.Email, "Reset Password",
516 $"Please reset your password by clicking here: <a href='{callbackUrl}'>link</a>");
517 return RedirectToAction(nameof(ForgotPasswordConfirmation));
518 }
519
520 // If we got this far, something failed, redisplay form
521 return View(model);
