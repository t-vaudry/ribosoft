253 
254  IList<Tuple<float, float>> currentResults;
255  string idealStructure;
256 
257  // Store distance and probability for further use, once we have the max distance
258  foreach (var d in designs)
259  {

261 
262  var foldOutputs = Fold(d.Sequence);
263 
264  idealStructure = d.IdealStructure;
265

270  if (status != R_STATUS.R_STATUS_OK)
271  {
272  throw new RibosoftAlgoException(status);
273  }
274 
275  currentResults.Add(new Tuple<float, float>(distance, output.Probability));
276  }
277 
278  structureResults.Add(currentResults);
279  }
280 
281  if (designs.Count != structureResults.Count)
282  {
283  throw new RibosoftAlgoException(R_STATUS.R_STRUCT_LENGTH_DIFFER);
284  }
285 
286  float maxDistance = structureResults.Max(results => results.Max(r => r.Item1));
287 
288  float currentStructureScore;
289 
290  for (int i = 0; i < designs.Count; i++)
291  {
292  currentStructureScore = 0.0f;
293 
294  foreach (Tuple<float, float> foldResults in structureResults[i])
295  {
296  currentStructureScore += (1 - (foldResults.Item1 / maxDistance)) * foldResults.Item2;
297  }
