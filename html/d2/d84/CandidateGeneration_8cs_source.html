212  public void GenerateStructure(List<List<Node>> nodesAtDepth, String inputSequence, String inputStructure, bool isRibozyme)
213  {
214  int depth = inputSequence.Length;
215  int latestNonRepeatIndex = -1;
216 
217  for (int i = 0; i < depth; i++)
218  {
219  //Ignore repeat notations for now. They will be dealt with at the end
220  if (!isRibozyme && RepeatStructureSymbols.Contains(inputStructure[i]))
221  {
222  continue;
223  }
224  latestNonRepeatIndex++;
225 
226  //List for nodes at current depth
227  List<Node> depth_i = new List<Node>();
228 
229  //Get input at index (nucleotide and secondary structure)
230  Nucleotide nucleotide = new Nucleotide(inputSequence[i]);
231 
232  //The neighbour (of a bond or pseudoknot)
233  int? neighbourIndex = null;
234  bool isTarget = false;
235  if (isRibozyme)
236  {
237  char structure = inputStructure[i];
238  isTarget = IsTarget(structure);
239  //If nucleotide is NOT a target
240  if (!isTarget)
241  {
242  //Determine if the nucleotide has a neighbour (bond or pseudoknot)
243  HasNeighbor(structure, i, ref neighbourIndex);
244  }
245  }
246 
247  List<char> baseList = isTarget && isRibozyme ? new List<char> { nucleotide.Symbol } : nucleotide.Bases;
248  foreach (char baseChar in baseList)
249  {
250  Node currentNode = new Node(new Nucleotide(baseChar), i, neighbourIndex);
251 
252  if (latestNonRepeatIndex > 0)
253  {
254  currentNode.Parents = nodesAtDepth[latestNonRepeatIndex - 1];
255  }
256 
257  //Add this node to the nodes at the current depth
258  depth_i.Add(currentNode);
259  }
260 
261  //Go to the previous depth and set all nodes' children to nodes of current depth
262  if (latestNonRepeatIndex > 0)
263  {
264  foreach (Node node in nodesAtDepth[latestNonRepeatIndex - 1])
265  {
266  node.Children = depth_i;
267  }
268  }
269 
270  //If we have found the second element of a neighbour pair, set the neighbour index of both nodes in pair
271  if (neighbourIndex.HasValue)
272  {
273  foreach (Node firstNeighbour in nodesAtDepth[neighbourIndex.Value])
274  {
275  firstNeighbour.NeighbourIndex = i;
276  }
277  }
278 
279  //Add all nodes created at this depth to the structue
280  nodesAtDepth.Add(depth_i);
281  }
282 
283  //Validate that all open parentheses have been closed
284  ValidateOpenBonds();
285  }

293  private void HasNeighbor(char structure, int i, ref int? neighbourIndex)
294  {
295  switch (structure)
296  {
297  case '.': //Nothing to do
298  break;
299  case '(': //Start an open bond
300  OpenBondIndices.Push(i);
301  break;
302  case ')': //Close an open bond
303  neighbourIndex = OpenBondIndices.Pop();
304  NeighboursIndices.Add(Tuple.Create(i, neighbourIndex.Value));
305  break;
306  case '[': //Start a pseudoknot
307  OpenPseudoKnotIndices.Push(i);
308  break;
309  case ']': //Close a pseudoknot
310  neighbourIndex = OpenPseudoKnotIndices.Pop();
311  NeighboursIndices.Add(Tuple.Create(i, neighbourIndex.Value));
312  break;
313  default: //Should not happen
314  throw new CandidateGenerationException("Unrecognized structure symbol encountered.");
315  }
316  }

346  public bool GetLargestSetSubstrateRegion()
347  {
348  int tmpStart = -1;
349  int tmpEnd = -1;
350 
351  int longestStart = -1;
352  int longestEnd = -1;
353  int longest = -1;
354 
355  bool started = false;
356 
357  //Only consider the portions without repeat regions
358  int start = NodesAtDepthCutSite[0][0].Depth;
359  int end = NodesAtDepthCutSite[NodesAtDepthCutSite.Count - 1][0].Depth;
360 
361  for(int i = start; i < end; i++)
362  {
363  char c = Ribozyme.SubstrateSequence[i];
364  if (!started && (c == 'A' || c == 'C' || c == 'G' || c == 'U'))
365  {
366  started = true;
367  tmpStart = i;
368  }
369  else if (started && !(c == 'A' || c == 'C' || c == 'G' || c == 'U'))
370  {
371  started = false;
372  tmpEnd = i;
373 
374  if (tmpEnd - tmpStart > longest)
375  {
376  longest = tmpEnd - tmpStart;
377  longestEnd = tmpEnd;
378  longestStart = tmpStart;
379  }
380  }
381  }
382 
383  bool success = longestStart != -1 && longestEnd != -1;
384 
385  if (success)
386  {
387  SmallestSubstarteInfo.StartIndex = longestStart;
388  SmallestSubstarteInfo.EndIndex = longestEnd;
389  SmallestSubstarteInfo.SubstrateBase = Ribozyme.SubstrateSequence.Substring(longestStart, longest);
390  }
391 
392  return success;
393  }

398  public void GetSubstrates()
399  {
400  //Locations of specified substrate portion within the RNA input
401  List<int> substrateIndices = AllIndicesOf(InputRNASequence, SmallestSubstarteInfo.SubstrateBase);
402 
403  //Number of characters to specify before and after the specified portion
404  int lengthBefore = SmallestSubstarteInfo.StartIndex - NodesAtDepthCutSite[0][0].Depth;
405  int lengthAfter = NodesAtDepthCutSite[NodesAtDepthCutSite.Count-1][0].Depth - SmallestSubstarteInfo.EndIndex + 1;
406 
407  //Total length of the specified portion
408  int length = SmallestSubstarteInfo.EndIndex - SmallestSubstarteInfo.StartIndex;
409 
410  //Loop through all spots in the RNA input where the specified portion of the substrate was found
411  //Complete the unspecified part of the substrate using the RNA input
412  foreach (int idxInRNA in substrateIndices)
413  {
414  Sequence sequence = new Sequence();
415 
416  //If the index in the RNA is such that the substrate won't fit around it, ignore it
417  if (idxInRNA < lengthBefore || (InputRNASequence.Length - idxInRNA) < lengthAfter + length)
418  continue;
419 
420  bool found = true;
421  //Add the elements before the specified portion
422  for (int i = lengthBefore; i > 0; i--)
423  {
424  Nucleotide substrate = new Nucleotide(Ribozyme.SubstrateSequence[SmallestSubstarteInfo.StartIndex - i]);
425 
426  if (substrate.Bases.Contains(InputRNASequence[idxInRNA - i]))
427  {
428  sequence.Nucleotides.Add(new Nucleotide(InputRNASequence[idxInRNA - i]));
429  }
430  else
431  {
432  found = false;
433  break;
434  }
435  }
436 
437  if (!found)
438  {
439  continue;
440  }
441 
442  //Add the specified portion
443  foreach (char c in SmallestSubstarteInfo.SubstrateBase)
444  {
445  sequence.Nucleotides.Add(new Nucleotide(c));
446  }
447 
448  //Add the elements after the specified portion
449  for (int i = 0; i < lengthAfter; i++)
450  {
451  Nucleotide substrate = new Nucleotide(Ribozyme.SubstrateSequence[SmallestSubstarteInfo.EndIndex + i]);
452 
453  if (substrate.Bases.Contains(InputRNASequence[idxInRNA + length + i]))
454  {
455  sequence.Nucleotides.Add(new Nucleotide(InputRNASequence[idxInRNA + length + i]));
456  }
457  else
458  {
459  found = false;
460  break;
461  }
462  }
463 
464  if (found)
465  {
466  //Add unique, since it is possible to build the same substrate multiple times from the RNA
467  SubstrateInfo substrateInfo = new SubstrateInfo(sequence, SubstrateBaseStructure);
468  if (!SubstrateInfo.Contains(substrateInfo))
469  SubstrateInfo.Add(substrateInfo);
470  }
471  }

504  public void TraverseSequence(Sequence currentSequence, Node currentNode)
505  {
506  currentSequence.Nucleotides.Add(currentNode.Nucleotide);
507 
508  //If leaf, add sequence to list
509  if (currentNode.Children.Count == 0) //Leaf
510  {
511  if (!Sequences.Contains(currentSequence))
512  {
513  Sequences.Add(currentSequence);
514  }
515  }
516  else
517  {
518  //If child is going to be linked to RNA, just continue
519  if (IsTarget(Ribozyme.Structure[currentNode.Depth + 1]))
520  {
521  Node child = new Node(currentNode.Children[0]);
522  child.Nucleotide.SetSymbol('-');
523  TraverseSequence(new Sequence(currentSequence), child);
524  }
525  //Else if the children have a neighbour (will all be the same, so just check 0) and that neighbour has already been set, we must choose
526  else if (currentNode.Children[0].NeighbourIndex.HasValue && currentNode.Children[0].NeighbourIndex.Value < currentNode.Depth + 1)
527  {
528  //Get the complement of the base at the specified neighbour index in the input string
529  char[] requiredBases = currentSequence.Nucleotides[currentNode.Children[0].NeighbourIndex.Value].GetSpecialComplements();
530  bool found = false;
531  foreach (char c in requiredBases)
532  {
533  foreach (Node child in currentNode.Children)
534  {
535  if (child.Nucleotide.Symbol == c)
536  {
537  TraverseSequence(new Sequence(currentSequence), child);
538  found = true;
539  break;
540  }
541  }
542  if (!found)
543  {
544  //If not found, there is the possibility that one of the neighbour's siblings is valid.
545  //In this case, there is no error: silently discard this sequence
546  Nucleotide inputNeighbourNucleotide = new Nucleotide(Ribozyme.Sequence[currentNode.Children[0].NeighbourIndex.Value]);
547  if (inputNeighbourNucleotide.Bases.Contains(c))
548  {
549  return;
550  }
551  }
552  }
553 
554  if (!found)
555  {
556  throw new CandidateGenerationException("Neighbours don't match!");
557  }
558  }
559  else
560  {
561  foreach (Node child in currentNode.Children)
562  TraverseSequence(new Sequence(currentSequence), child);
563  }
564  }
565  }

594  public void GetRepeatNotationInfo()
595  {
596  StringBuilder baseStructure = new System.Text.StringBuilder();
597 
598  bool previousIsRepeat = false;
599  int currentRegionStartIndex = -1;
600 
601  //Check if any of the inputs are lowercase (repeat notation)
602  for (int i = 0; i < Ribozyme.SubstrateSequence.Length; i++)
603  {
604  if (char.IsLower(Ribozyme.SubstrateSequence[i]))
605  {
606  char structureSymbol = Ribozyme.SubstrateStructure[i];
607  if (!IsTarget(structureSymbol))
608  {
609  throw new CandidateGeneration.CandidateGenerationException("Repeat notation is not on target. Unhandled case.");
610  }
611 
612  RepeatStructureSymbols.Add(structureSymbol);
613 
614  if (!previousIsRepeat)
615  {
616  //Start new repeat region
617  currentRegionStartIndex = i;
618  }
619 
620  previousIsRepeat = true;
621  }
622  else
623  {
624  baseStructure.Append(Ribozyme.SubstrateStructure[i]);
625 
626  if (previousIsRepeat)
627  {
628  //End repeat region
629  RepeatRegions.Add(Tuple.Create(currentRegionStartIndex, i - 1));
630  previousIsRepeat = false;
631  }
632  }
633  }
634 
635  //Close the last repeat region, if any
636  if (previousIsRepeat)
637  {
638  RepeatRegions.Add(Tuple.Create(currentRegionStartIndex, Ribozyme.SubstrateSequence.Length - 1));
639  }
640 
641  //Check if multiple repeat regions. This case isn't handled yet
642  if (RepeatRegions.Count > 2)
643  {
644  throw new CandidateGenerationException("More than 2 repeat regions are not supported.");
645  }
646 
647  //Check if any repeat regions aren't at either extremity. This case isn't handled yet.
648  foreach(Tuple<int,int> repeatRegion in RepeatRegions)
649  {
650  //If start at the beginning, good
651  if (repeatRegion.Item1 == 0)
652  {
653  continue;
654  }
655 
656  //If end at the end, good
657  if (repeatRegion.Item2 == Ribozyme.SubstrateSequence.Length-1)
658  {
659  continue;
660  }
661 
662  //The repeat region is in the middle of the sequence. We don't handle this right now.
663  throw new CandidateGenerationException("Repeat notation not located at beginning or end of substrate sequence. Case not supported.");
664  }
665 
666  SubstrateBaseStructure = baseStructure.ToString();
667  }

673  public IEnumerable<Candidate> CompleteSequencesWithCutSiteInfo()
674  {
675  //First, build the mapping between substrate and ribozyme
676  for (int i = 0; i < Ribozyme.SubstrateStructure.Length; i++)
677  {
678  //If the substrate is not part of the target, continue
679  if (Ribozyme.SubstrateStructure[i] == '.')
680  {
681  continue;
682  }
683 
684  //If it is not a '.', we are expecting it to be part of the target
685  if (!IsTarget(Ribozyme.SubstrateStructure[i]))
686  {
687  throw new CandidateGenerationException(String.Format("Unexpected substrate structure character: {0}", Ribozyme.SubstrateStructure[i]));
688  }
689 
690  //Now find the base in the input sequence that binds to this (aka has the same struc value)
691  bool foundMatch = false;
692 
693  for (int j = 0; j < Ribozyme.Structure.Length; j++)
694  {
695  if (Ribozyme.Structure[j] == Ribozyme.SubstrateStructure[i])
696  {
697  foundMatch = true;
698  RibozymeSubstrateIndexPairs.Add(Tuple.Create(j, i));
699  }
700  }
701 
702  if (!foundMatch)
703  {
704  throw new CandidateGenerationException(String.Format("Substrate structure character not found in ribozyme structure: {0}", Ribozyme.SubstrateStructure[i]));
705  }
706  }
707 
708  //Complete each sequence with the complement of the substrate at the target positions
709  foreach (SubstrateInfo substrateInfo in SubstrateInfo)
710  {
711  String substrateComplement = substrateInfo.Sequence.GetComplement();
712 
713  foreach (Sequence ribozymeSequence in Sequences)
714  {
715  Sequence newSequence = new Sequence(ribozymeSequence);
716  bool success = true;
717 
718  //For each element in the ribozyme sequence that is part of the target area, check if it is possible to bond with the substrate
719  foreach (Tuple<int, int> indexPair in RibozymeSubstrateIndexPairs)
720  {
721  int riboIdx = indexPair.Item1;
722  int substrateIdx = indexPair.Item2;
723 
724  //Check if this substrate sequence has this bond (may not due to repeat notation)
725  char bondID = Ribozyme.SubstrateStructure[substrateIdx];
726  substrateIdx = substrateInfo.Structure.IndexOf(bondID);
727  if (substrateIdx == -1)
728  {
729  continue;
730  }
731 
732  Nucleotide ribozymeNucleotide = new Nucleotide(Ribozyme.Sequence[riboIdx]);
733  if (ribozymeNucleotide.Bases.Contains(substrateComplement[substrateIdx]))
734  {
735  newSequence.Nucleotides[riboIdx] = new Nucleotide(substrateComplement[substrateIdx]);
736  }
737  else
738  {
739  success = false;
740  break;
741  }
742  }
743 
744  //If all target elements can successfully bond, add this sequence to the list
745  if (success)
746  {
747  //Remove the nucleotides that are not part of the sequence (due to repeat notation)
748  String newStructure = Ribozyme.Structure;
749  RemoveUnusedRepeats(ref newStructure, newSequence);
750  yield return new Candidate { Sequence = newSequence, Structure = newStructure, SubstrateSequence = substrateInfo.Sequence.GetString(), SubstrateStructure = substrateInfo.Structure, CutsiteNumberOffset = substrateInfo.CutsiteOffset, CutsiteIndices = AllIndicesOf(InputRNASequence, substrateInfo.Sequence.GetString()) };
751  }
752  //Else, do nothing: this ribozyme sequence cannot bond with the substrate
753  }
754  }
755  }

777  public void HandleExtremityRepeats()
778  {
779  foreach (Tuple<int, int> repeatRegion in RepeatRegions)
780  {
781  //Always base off of the current list
782  //If this isn't the first repeat region, this will base off of the no repeat versions as well as all the ones with the other repeat region
783  IList<SubstrateInfo> sequencesToBaseOffOf = new List<SubstrateInfo>(SubstrateInfo);
784 
785  int currentCount = 0;
786 
787  IList<SubstrateInfo> newSubstrateSequences = new List<SubstrateInfo>();
788 
789  bool startRepeat = (repeatRegion.Item1 == 0);
790  bool endRepeat = (repeatRegion.Item2 == Ribozyme.SubstrateSequence.Length - 1);
791 
792  while (currentCount <= (repeatRegion.Item2 - repeatRegion.Item1))
793  {
794  int beginIdx = -1;
795 
796  if (!startRepeat && !endRepeat)
797  {
798  throw new CandidateGenerationException("Repeat notation not located at beginning or end of substrate sequence. Case not supported.");
799  }
800 
801  beginIdx = startRepeat ? (repeatRegion.Item2 - currentCount) : (repeatRegion.Item1 + currentCount);
802 
803  Nucleotide newBeginning = new Nucleotide(Ribozyme.SubstrateSequence[beginIdx]);
804  char additionalStructure = Ribozyme.SubstrateStructure[beginIdx];
805 
806  foreach (char baseSymbol in newBeginning.Bases)
807  {
808  foreach (SubstrateInfo seq in sequencesToBaseOffOf)
809  {
810  Sequence newSeq = new Sequence(seq.Sequence);
811  int insertIdx = startRepeat ? 0 : newSeq.GetLength();
812  newSeq.Insert(insertIdx, new Nucleotide(baseSymbol));
813  String newStructure = startRepeat ? additionalStructure + seq.Structure : seq.Structure + additionalStructure;
814 
815  int offset = startRepeat ? (currentCount + 1) : 0; //If there is a repeat at the beginning, the cutsite number should be shifted
816 
817  //Keep only cutsites that are found in RNA
818  if (InputRNASequence.IndexOf(newSeq.GetString(), StringComparison.Ordinal) >= 0)
819  {
820  newSubstrateSequences.Add(new SubstrateInfo(newSeq, newStructure, offset));
821  }
822  }
823  }
824 
825  //Add valid cutsites to list
826  SubstrateInfo.AddRange(newSubstrateSequences);
827 
828  //Next added repeat will base itself off of this new list
829  sequencesToBaseOffOf = new List<SubstrateInfo>(newSubstrateSequences);
830  newSubstrateSequences.Clear();
831 
832  currentCount++;
833  }
834  }

870  public static List<int> AllIndicesOf(string str, string value)
871  {
872  if (String.IsNullOrEmpty(value))
