616  private void DownloadFiles(IQueryable<Design> designs, JObject obj, string format, out MemoryStream stream, out string extension, out string type)
617  {
618  var payload = "";
619  stream = new MemoryStream();
620  byte[] byteArray;
621  switch (format)
622  {
623  case "csv":
624  payload += String.Format("Rank,DesiredTemperatureScore,HighestTemperatureScore,SpecificityScore,AccessibilityScore,StructureScore,CreatedAt,UpdatedAt,Sequence\n");
625  extension = "csv";
626  foreach (Design d in designs)
627  {
628  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
629  {
630  payload += String.Format("{0},{1},{2},{3},{4},{5},{6},{7},{8}\n", d.Rank, d.DesiredTemperatureScore, d.HighestTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
631  }
632  }
633  type = "application/csv";
634  byteArray = Encoding.ASCII.GetBytes(payload);
635  stream = new MemoryStream(byteArray);
636  break;
637  case "zip":
638  extension = "zip";
639  type = "application/octet-stream";
640  ZipOutputStream zipStream = new ZipOutputStream(stream);
641  foreach (Design d in designs)
642  {
643  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
644  {
645  AddToZip(ref zipStream, d);
646  }
647  }
648 
649  zipStream.IsStreamOwner = false;
650  zipStream.Close();
651  stream.Position = 0;
652  break;
653  case "fasta":
654  extension = "fasta";
655  foreach (Design d in designs)
656  {
657  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
658  {
659  payload += String.Format(">Rank {0} | DesiredTemperatureScore {1} | HighestTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.DesiredTemperatureScore, d.HighestTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
660  }
661  }
662  type = "text/plain";
663  byteArray = Encoding.ASCII.GetBytes(payload);
664  stream = new MemoryStream(byteArray);
665  break;
666  default:
667  extension = "";
668  type = "";
669  break;
670  }
