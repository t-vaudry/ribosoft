181 public async Task<IActionResult> Details(int? id, string sortOrder, int pageNumber, string filterData)
182 {
183 if (id == null)
184 {
185 return NotFound();
186 }
187
188 var user = await GetUser();
189
190 var job = await _context.Jobs
191 .Include(j => j.Owner)
192 .Include(j => j.Ribozyme)
193 .Include(j => j.Assembly)
194 .Where(j => j.OwnerId == user.Id)
195 .SingleOrDefaultAsync(m => m.Id == id);
196
197 if (job == null)
198 {
199 return NotFound();
200 }
201
202 var designs = from d in _context.Designs where d.JobId == job.Id select d;
203
204 List<Filter> filterList = new List<Filter>();
205 if (!string.IsNullOrEmpty(filterData))
206 {
207 filterData = Encoding.UTF8.GetString(Convert.FromBase64String(filterData));
208 var deserializedFilters = JsonConvert.DeserializeObject<List<Filter>>(filterData);
209 if (deserializedFilters != null)
210 {
211 filterList = deserializedFilters;
212 }
213 foreach (var filter in filterList)
214 {
215 FilterDesigns(ref designs, filter.param, filter.condition, float.Parse(filter.value));
216 }
217 }
218
219 int pageSize = 20;
220 pageNumber = Math.Max(pageNumber, 1);
221 int offset = (pageSize * pageNumber) - pageSize;
222
223 SetSortParams(sortOrder);
224 SortDesigns(ref designs, sortOrder);
225
226 var vm = new JobDetailsViewModel();
227
228 vm.Job = job;
229 vm.SortOrder = sortOrder;
230 vm.FilterList = filterList;
231 vm.Designs.Data = await designs.Skip(offset).Take(pageSize).AsNoTracking().ToListAsync();
232 vm.Designs.TotalItems = await designs.CountAsync();
233 vm.Designs.PageNumber = pageNumber;
234 vm.Designs.PageSize = pageSize;
235
236 return View(vm);
237 }

700 private void DownloadFiles(IQueryable<Design> designs, JObject? obj, string format, out MemoryStream stream, out string extension, out string type)
701 {
702 var payload = "";
703 stream = new MemoryStream();
704 byte[] byteArray;
705 switch (format)
706 {
707 case "csv":
708 payload += String.Format("Rank,CutsiteIndex,DesiredTemperatureScore,SpecificityScore,AccessibilityScore,StructureScore,CreatedAt,UpdatedAt,Sequence\n");
709 extension = "csv";
710 foreach (Design d in designs)
711 {
712 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
713 {
714 payload += String.Format("{0},{1},{2},{3},{4},{5},{6},{7},{8}\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
715 }
716 }
717 type = "application/csv";
718 byteArray = Encoding.ASCII.GetBytes(payload);
719 stream = new MemoryStream(byteArray);
720 break;
721 case "zip":
722 extension = "zip";
723 type = "application/octet-stream";
724 ZipOutputStream zipStream = new ZipOutputStream(stream);
725 foreach (Design d in designs)
726 {
727 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
728 {
729 AddToZip(ref zipStream, d);
730 }
731 }
732
733 zipStream.IsStreamOwner = false;
734 zipStream.Close();
735 stream.Position = 0;
736 break;
737 case "fasta":
738 extension = "fasta";
739 foreach (Design d in designs)
740 {
741 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
742 {
743 payload += String.Format(">Rank {0} | CutsiteIndex {1} | DesiredTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
744 }
745 }
746 type = "text/plain";
747 byteArray = Encoding.ASCII.GetBytes(payload);
748 stream = new MemoryStream(byteArray);
749 break;
750 default:
751 extension = "";
752 type = "";
753 break;
754 }

762 private void AddToZip(ref ZipOutputStream zipStream, Design d)
763 {
764 byte[] byteArray;
765 var newEntry = new ZipEntry("design" + d.Id + ".fasta");
766 newEntry.DateTime = DateTime.Now;
767
768 zipStream.PutNextEntry(newEntry);
769 byteArray = Encoding.ASCII.GetBytes(String.Format(">Rank {0} | CutsiteIndex {1} | DesiredTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence));
770
771 MemoryStream inStream = new MemoryStream(byteArray);
772 StreamUtils.Copy(inStream, zipStream, new byte[4096]);
773 inStream.Close();
774 zipStream.CloseEntry();
