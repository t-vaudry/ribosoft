175 public async Task<IActionResult> Details(int? id, string sortOrder, int pageNumber, string filterData)
176 {
177 if (id == null)
178 {
179 return NotFound();
180 }
181
182 var user = await GetUser();
183
184 var job = await _context.Jobs
185 .Include(j => j.Owner)
186 .Include(j => j.Ribozyme)
187 .Include(j => j.Assembly)
188 .Where(j => j.OwnerId == user.Id)
189 .SingleOrDefaultAsync(m => m.Id == id);
190
191 if (job == null)
192 {
193 return NotFound();
194 }
195
196 var designs = from d in _context.Designs where d.JobId == job.Id select d;
197
198 List<Filter> filterList = new List<Filter>();
199 if (!string.IsNullOrEmpty(filterData))
200 {
201 filterData = Encoding.UTF8.GetString(Convert.FromBase64String(filterData));
202 filterList = JsonConvert.DeserializeObject<List<Filter>>(filterData);
203 foreach (var filter in filterList)
204 {
205 FilterDesigns(ref designs, filter.param, filter.condition, float.Parse(filter.value));
206 }
207 }
208
209 int pageSize = 20;
210 pageNumber = Math.Max(pageNumber, 1);
211 int offset = (pageSize * pageNumber) - pageSize;
212
213 SetSortParams(sortOrder);
214 SortDesigns(ref designs, sortOrder);
215
216 var vm = new JobDetailsViewModel();
217
218 vm.Job = job;
219 vm.SortOrder = sortOrder;
220 vm.FilterList = filterList;
221 vm.Designs.Data = await designs.Skip(offset).Take(pageSize).AsNoTracking().ToListAsync();
222 vm.Designs.TotalItems = await designs.CountAsync();
223 vm.Designs.PageNumber = pageNumber;
224 vm.Designs.PageSize = pageSize;
225
226 return View(vm);
227 }

685 private void DownloadFiles(IQueryable<Design> designs, JObject obj, string format, out MemoryStream stream, out string extension, out string type)
686 {
687 var payload = "";
688 stream = new MemoryStream();
689 byte[] byteArray;
690 switch (format)
691 {
692 case "csv":
693 payload += String.Format("Rank,CutsiteIndex,DesiredTemperatureScore,SpecificityScore,AccessibilityScore,StructureScore,CreatedAt,UpdatedAt,Sequence\n");
694 extension = "csv";
695 foreach (Design d in designs)
696 {
697 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
698 {
699 payload += String.Format("{0},{1},{2},{3},{4},{5},{6},{7},{8}\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
700 }
701 }
702 type = "application/csv";
703 byteArray = Encoding.ASCII.GetBytes(payload);
704 stream = new MemoryStream(byteArray);
705 break;
706 case "zip":
707 extension = "zip";
708 type = "application/octet-stream";
709 ZipOutputStream zipStream = new ZipOutputStream(stream);
710 foreach (Design d in designs)
711 {
712 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
713 {
714 AddToZip(ref zipStream, d);
715 }
716 }
717
718 zipStream.IsStreamOwner = false;
719 zipStream.Close();
720 stream.Position = 0;
721 break;
722 case "fasta":
723 extension = "fasta";
724 foreach (Design d in designs)
725 {
726 if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
727 {
728 payload += String.Format(">Rank {0} | CutsiteIndex {1} | DesiredTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
729 }
730 }
731 type = "text/plain";
732 byteArray = Encoding.ASCII.GetBytes(payload);
733 stream = new MemoryStream(byteArray);
734 break;
735 default:
736 extension = "";
737 type = "";
738 break;
739 }

747 private void AddToZip(ref ZipOutputStream zipStream, Design d)
748 {
749 byte[] byteArray;
750 var newEntry = new ZipEntry("design" + d.Id + ".fasta");
751 newEntry.DateTime = DateTime.Now;
752
753 zipStream.PutNextEntry(newEntry);
754 byteArray = Encoding.ASCII.GetBytes(String.Format(">Rank {0} | CutsiteIndex {1} | DesiredTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.CutsiteIndex, d.DesiredTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence));
755
756 MemoryStream inStream = new MemoryStream(byteArray);
757 StreamUtils.Copy(inStream, zipStream, new byte[4096]);
758 inStream.Close();
759 zipStream.CloseEntry();
