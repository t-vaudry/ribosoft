81 public IList<Database> GetAvailableDatabases(string path)
82 {
83 var args = string.Format("-list {0} -recursive -list_outfmt \"%f\t%t\t%d\t%l\t%n\t%U\"", EncodeParameterArgument(path));
84
85 var process = new Process
86 {
87 StartInfo = new ProcessStartInfo
88 {
89 FileName = "blastdbcmd",
90 Arguments = args,
91 RedirectStandardOutput = true,
92 UseShellExecute = false,
93 CreateNoWindow = true
94 }
95 };
96
97 var databases = new List<Database>();
98 var tagRegex = new Regex(@"\[(?<name>[^=]+)=(?<value>[^]]+)\]+");
99
100 // start proc
101 process.Start();
102
103 // read output
104 string outputLine;
105 while ((outputLine = process.StandardOutput.ReadLine()) != null)
106 {
107 if (string.IsNullOrWhiteSpace(outputLine))
108 {
109 continue;
110 }
111
112 var columns = outputLine.Split('\t');
113
114 if (columns.Length != 6)
115 {
116 continue;
117 }
118
119 var database = new Database
120 {
121 AbsolutePath = columns[0],
122 RelativePath = columns[0].Substring(path.Length + 1),
123 UpdatedAt = DateTime.Parse(columns[2]),
124 Nucleotides = BigInteger.Parse(columns[3]),
125 Sequences = BigInteger.Parse(columns[4]),
126 Bytes = BigInteger.Parse(columns[5])
127 };
128
129 var tagMatches = tagRegex.Matches(columns[1]);
130
131 if (tagMatches.Count < 6)
132 {
133 continue;
134 }
135
136 // parse out [name=value] tags in the db name to collect metadata
137 foreach (Match match in tagMatches)
138 {
139 switch (match.Groups["name"].Value.ToLowerInvariant())
140 {
141 case "assembly_accession":
142 database.AccessionId = match.Groups["value"].Value;
143 break;
144 case "asm_name":
145 database.AssemblyName = match.Groups["value"].Value;
146 break;
147 case "taxid":
148 database.TaxonomyId = int.Parse(match.Groups["value"].Value);
149 break;
150 case "species_taxid":
151 database.SpeciesTaxonomyId = int.Parse(match.Groups["value"].Value);
152 break;
153 case "organism_name":
154 database.OrganismName = match.Groups["value"].Value;
155 break;
156 case "type":
157 database.Type = match.Groups["value"].Value;
158 break;
159 }
160 }
161
162 databases.Add(database);
163 }
164
165 process.WaitForExit();
166
167 // cleanup resources
168 process.Close();
169
170 return databases;
171 }

229 public string BuildArgumentString(BlastParameters parameters)
230 {
231 var builder = new StringBuilder();
232
233 // task
234 string task;
235
236 switch(parameters.Task)
237 {
238 case BlastParameters.BlastTask.blastn:
239 task = "blastn";
240 break;
241 case BlastParameters.BlastTask.blastn_short:
242 task = "blastn-short";
243 break;
244 case BlastParameters.BlastTask.dc_megablast:
245 task = "dc-megablast";
246 break;
247 case BlastParameters.BlastTask.rmblastn:
248 task = "rmblastn";
249 break;
250 case BlastParameters.BlastTask.megablast:
251 default:
252 task = "megablast";
253 break;
254 }
255
256 builder.AppendFormat("-task {0} ", task);
257
258 // db
259 builder.AppendFormat("-db {0} ", EncodeParameterArgument(parameters.Database));
260
261 // dust
262 builder.AppendFormat("-dust {0} ", parameters.Dust ? "yes" : "no");
263
264 // soft_masking
265 builder.AppendFormat("-soft_masking {0} ", parameters.SoftMasking ? "true" : "false");
266
267 // max_target_seqs
268 builder.AppendFormat("-max_target_seqs {0} ", parameters.MaxTargetSequences);
269
270 // use_index
271 builder.AppendFormat("-use_index {0} ", parameters.UseIndex ? "true" : "false");
272
273 // num_threads
274 builder.AppendFormat("-num_threads {0} ", Math.Max(Math.Min(parameters.NumThreads, 8), 1));
275
276 // outfmt
277 builder.AppendFormat("-outfmt {0} ", EncodeParameterArgument(parameters.OutputFormat));
278
279 // evalue
280 builder.AppendFormat("-evalue {0} ", parameters.ExpectValue);
281
282 return builder.ToString();
283 }

290 public static string EncodeParameterArgument(string original)
291 {
292 if (string.IsNullOrEmpty(original))
293 {
294 return original;
295 }
296
297 string value = Regex.Replace(original, @"(\\*)" + "\"", @"$1\$0");
298 value = Regex.Replace(value, @"^(.*\s.*?)(\\*)$", "\"$1$2$2\"");
299
300 return value;
301 }
