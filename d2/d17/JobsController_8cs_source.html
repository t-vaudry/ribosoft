588  private void DownloadFiles(IQueryable<Design> designs, JObject obj, string format, out MemoryStream stream, out string extension, out string type)
589  {
590  var payload = "";
591  stream = new MemoryStream();
592  byte[] byteArray;
593  switch (format)
594  {
595  case "csv":
596  payload += String.Format("Rank,DesiredTemperatureScore,HighestTemperatureScore,SpecificityScore,AccessibilityScore,StructureScore,CreatedAt,UpdatedAt,Sequence\n");
597  extension = "csv";
598  foreach (Design d in designs)
599  {
600  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
601  {
602  payload += String.Format("{0},{1},{2},{3},{4},{5},{6},{7},{8}\n", d.Rank, d.DesiredTemperatureScore, d.HighestTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
603  }
604  }
605  type = "application/csv";
606  byteArray = Encoding.ASCII.GetBytes(payload);
607  stream = new MemoryStream(byteArray);
608  break;
609  case "zip":
610  extension = "zip";
611  type = "application/octet-stream";
612  ZipOutputStream zipStream = new ZipOutputStream(stream);
613  foreach (Design d in designs)
614  {
615  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
616  {
617  AddToZip(ref zipStream, d);
618  }
619  }
620 
621  zipStream.IsStreamOwner = false;
622  zipStream.Close();
623  stream.Position = 0;
624  break;
625  case "fasta":
626  extension = "fasta";
627  foreach (Design d in designs)
628  {
629  if (obj == null || obj.ContainsKey(d.JobId.ToString() + '-' + d.Id.ToString()))
630  {
631  payload += String.Format(">Rank {0} | DesiredTemperatureScore {1} | HighestTemperatureScore {2} | SpecificityScore {3} | AccessibilityScore {4} | StructureScore {5} | CreatedAt {6} | UpdatedAt {7}\n{8}\n\n", d.Rank, d.DesiredTemperatureScore, d.HighestTemperatureScore, d.SpecificityScore, d.AccessibilityScore, d.StructureScore, d.CreatedAt, d.UpdatedAt, d.Sequence);
632  }
633  }
634  type = "text/plain";
635  byteArray = Encoding.ASCII.GetBytes(payload);
636  stream = new MemoryStream(byteArray);
637  break;
638  default:
639  extension = "";
640  type = "";
641  break;
642  }
