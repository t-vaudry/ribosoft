290  private async Task RunBlast(Job job, IJobCancellationToken cancellationToken)
291  {
292  // check if blastn is available; if it isn't, ignore specificity
293  if (!_blaster.IsAvailable())
294  {
295  _logger.LogWarning("RibosoftWarning | BLAST Service is not available!!");
296  return;
297  }
298 
299  var designs = _db.Designs
300  .Where(d => d.JobId == job.Id)
301  .GroupBy(d => new { d.CutsiteIndex, d.SubstrateSequenceLength });
302 
303  foreach (var designGroup in designs)
304  {
305  cancellationToken.ThrowIfCancellationRequested();
306 
307  var design = designGroup.First();
308  var substrateSequence = job.RNAInput.Substring(design.CutsiteIndex, design.SubstrateSequenceLength);
309 
310  if (string.IsNullOrEmpty(substrateSequence))
311  {
312  continue;
313  }
314 
315  // calculate the substrate specificity score, which is common to all designs in this group
316  var substrateSpecificityScore = CalculateSpecificity(substrateSequence, job.Assembly.Path);
317 
318  if (job.SpecificityMethod == SpecificityMethod.CleavageAndHybridization)
319  {
320  // if we're doing hybridization specificity, also compute the score for the design sequence complement
321  foreach (var d in designGroup)
322  {
323  d.SpecificityScore = substrateSpecificityScore + CalculateSpecificity(new Sequence(d.Sequence).GetComplement(), job.Assembly.Path);
324  }
325  }
326  else
327  {
328  // for cleavage-only specificity, only use the substrate specificity score
329  foreach (var d in designGroup)
330  {
331  d.SpecificityScore = substrateSpecificityScore;
332  }
333  }
334  }
335 
336  await _db.SaveChangesAsync();
